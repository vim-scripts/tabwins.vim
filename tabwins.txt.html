 37 Section 1: Introduction                         TabwinsIntroduction
 38 ====================================================================
 39 
 40 Easy access to the power of the Vim GUI.
 41 
 42 See the screen shots at
 43   https://github.com/cesheridan/graphics/tree/master/tabwins
 44 
 45 
 46 1 command for immediate access to customized window structures
 47 and content ~during~ Vim sessions.
 48 
 49 The Developer
 50   - Avoids hit-or-miss, time-consuming, manual, repetitious
 51     runs of :split/:vsplit, and thought-sinks about how
 52     to produce the intended window structure
 53 
 54   - Avoids losing current configurations due to mid-session
 55     reloads of backed-up Vim sessions that have preferred
 56     window structures
 57 
 58   - Avoids loss of focus from these discontinuities
 59 
 60 
 61 FORMS --
 62 
 63   :{Tabwins cmd}   {window_dimensions}; {fill_spec,...};  {tab_option, ...}
 64   :{symmetric cmd}                      {fill_spec,...};  {tab_option, ...}
 65 
 66   Where
 67      :{Tabwins cmd} is any of :Tabwins, :TabwinsVertical, :TabwinsHorizontal
 68      Aliases                  :Tw,      :V,             & :H
 69      About aliases, see tabwins.txt section "Configuration"/TabwinsConfiguration.
 70 
 71      :{symmetric cmd} is any of the form :{V|H}IxJ or :{V|H}I 
 72 
 73      'fill_spec' 
 74      is a file, dir, shell/vim var, or string evaluable
 75      by :execute -- see EXAMPLES
 76 
 77      'tab_option'
 78      has Vim key:value pair syntax -- see EXAMPLES
 79 
 80   Semicolons separate arg groups, commas separate inside groups.
 81 
 82   {window_dimensions} is the only mandatory arg, for :{Tabwins cmd}
 83 
 84   SECTION "Usage/"TabwinsUsage IN tabwins.txt HAS MORE DETAIL.
 85 
 86 
 87 EXAMPLES --
 88 
 89   >> ----------------------------------------------------------------
 90   >> Empty Windows
 91   >> ----------------------------------------------------------------
 92 
 93   :Tabwins 1211
 94   :T       1211
 95   " For a new tab of 4 columns of 1, 2, 1, & 1 windows successively
 96   " left to right.
 97 
 98   :TabwinsHorizontal 4221
 99   :H                 4221
100   " For a new tab of 4 rows, with 4,2,2, & 1 windows successively
101   " top to bottom.
102 
103   >> ----------------------------------------------------------------
104   >> Populated Windows
105   >> ----------------------------------------------------------------
106 
107   :Tabwins            1211; $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'
108   :T                  {                   as above                        }
109   " 'fill_content' specs of files are automatically opened via :edit,
110   " while specs of dirs are opened via :Explore
111   " (Specify the '' empty string for windows intended to be empty.)
112 
113   :Tabwins           11111; $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'
114   :T                 {                    as above                        }
115   " Same buffer content, in different window structure, of 5 cols,
116   " one window in each.
117 
118   :TabwinsHorizontal 11111; $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'
119   :H                 {                    as above                        }
120   " Flip axis priority of above to the horizontal
121 
122   :Tabwins             312; '~/.bashrc', '~/.vimrc', '~/.git', $HOME, '/', '/usr/bin'
123   :T                   {                  as above                                  }
124 
125   >> ----------------------------------------------------------------
126   >> Symmetric Window Structures
127   >> ----------------------------------------------------------------
128   In a symmetric window, all secondary axis dimensions are the same.
129 
130   :V5x8
131   :H5x8
132   :V6
133   :H6
134   :V8x14
135   :H8x14
136   " Empty windows
137 
138   :V5x1
139 
140   :V5x1 $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'
141   " Structure & content same as :Tabwins 11111 above
142 
143   :V5   $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'
144   " Same result again: since the secondary axis has only 1
145   " window for its dimensions, V5 is available as an even
146   " shorter alias.
147 
148   >> ----------------------------------------------------------------
149   >> Dimensions >9
150   >> ----------------------------------------------------------------
151 
152   :Tabwins [ 5, 8, 13, 21 ] ~
153   :T       [ 5, 8, 13, 21 ]
154   " For a 4-column tab with 5, 8, 13, 21 windows successively left to right.
155 
156   :Tabwins [ 14, 14 ]
157   :T       [ 14, 14 ]
158   " Or, :V2x14
159 
160   :Tabwins [ 40, 30, 40, 30 ]
161   :T       [ 40, 30, 40, 30 ]
162   " For large monitors ...
163 
164 
165   >> ----------------------------------------------------------------
166   >> fill_specs with Multiple Commands
167   >> ----------------------------------------------------------------
168 
169   :Tabwins 1112;  $HOME, '~/.bashrc', '~/.vimrc', '/', 'Explore /usr/bin | resize 20'
170   :T       {                           as above                                     }
171   " Does a :resize of the last window, via Vim '|' command concat.  If a file
172   " or dir is followed with concated commands, the caller needs to explicitly
173   " add the command that opens the file or dir, i.e. 'Explore' in this example.
174 
175   >> ----------------------------------------------------------------
176   >> Tab Options
177   >> ----------------------------------------------------------------
178 
179   :Tabwins 1211;  $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'; 'ending_window_number' : 2
180   :Tabwins 1211;  $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'; 'netrw_liststyle'      : 3
181   :T              {                    as above                                             }
182 
183   :Tabwins 1211;  $HOME, '~/.bashrc', '~/.vimrc', '/', '/usr/bin'; 'netrw_liststyle'      : 3, 'ending_window_number' : 2
184   :T              {                    as above                                                                         }
185   " Multiple tab_options in same command.

188 TABWINS MENU --
189 
190   Default settings load Vim menu Tabwins, for access to selected
191                         ~~~~~~~~~~~~~~~~
192   tab-builder commands.  Those and additional tab builders are
193   available via Vim commands defined in tabwins.vim. The menu has
194   only a small portion of the tab builders defined in tabwins.vim.

227 Section 2: Usage                                       TabwinsUsage
228 ====================================================================
229 
230 => See FORMS    in TabwinsIntroduction
231 => See EXAMPLES in TabwinsIntroduction
232 
233 
234 THREE TABWINS COMMANDS --
235 
236   These commands all take the same arguments.
237 
238   :Tabwins
239     Same as :TabwinsVertical
240 
241   :TabwinsVertical
242     Gives priority to the vertical axis.
243 
244   :TabwinsHorizontal 
245     Gives priority to the horizontal axis.
246 
247   See "Vertical & Horizontal"/TabwinsVandH about primary/secondary
248   axes and vertical/horizontal priority.
249 
250   These commands accept arbitrary dimension specification,
251   unconstrained by configuration vars, other than
252   windows-per-tabsize limits of the Vim instance.
253 
254 
255 SYMMETRIC WINDOW STRUCTURES --
256 
257   In these, the lengths of the secondary axis dimensions
258   are all equal.
259 
260   Symmetric commands are available to reduce keystrokes.
261 
262   'H' indicates horizontal priority, 'V' vertical priority.
263 
264   Loading tabwins.vim creates :{V|H}IxJ commands where the
265   set of IxJ is the permutations of:
266     g:tabwins_max_v_size 
267     g:tabwins_max_h_size 
268 
269   The same global vars determine the set of :{V|H}J commands
270   also produced on tabwins.vim load, where N is from 1 to the
271   value of the relevant g:tabwins_max_{h|v}_size var above.
272 
273 
274 DIMENSIONS GREATER THAN 9 --
275 
276   Dimensions >9 are unparsable when {window_dimensions} is
277   an integer.  To enable dimensions >9, use list syntax as
278   shown in EXAMPLES of "Introduction"/TabwinsIntroduction
279 
280   List syntax applies to any tab that one or more dimension >9.
281   All the dimensions of such a tab are specified in one list.
282 
283 
284 
285 FILL SPECS WITH MULTIPLE COMMANDS --
286 
287   Multiple Vim commands can be specified in a fill_spec by
288   prepending commands with the Vim '|' concatenation operator.
289   See EXAMPLES in "Introduction"/TabwinsIntroduction.
290 
291 
292 TAB OPTIONS --
293 
294   Available tab_options are below, with global default vars
295   on the right.
296 
297   See tabwins.txt section "Configuration"/TabwinsConfiguration
298   re the globals.
299 
300   These options apply to the entire tab and can be combined when
301   separated by commas.
302 
303      'dir_cmd_name'              : g:tabwins_dir_command_name_default,
304      'netrw_liststyle'           : g:tabwins_netrw_liststyle_default,
305      'line_number_at_window_top' : g:tabwins_netrw_line_number_default,
306 
307      => Use a fill_spec for window-specific override of the
308         above options
309 
310      'first_window_number'       : g:tabwins_first_window_number_default,
311      => window population starts with this number, so you wouldn't
312         have to have to have a fill_spec for lower-numbered windows.
313 
314      'ending_window_number'      : g:tabwins_ending_window_number_default,
315      => the active window number after tabwins.vim builds a tab
316 
317 
318   See EXAMPLES in "Introduction"/TabwinsIntroduction.

322 ====================================================================
323 Section 3: Vertical & Horizontal                    TabwinsVandH
324 ====================================================================
325 
326 Windows are built one axis at a time. The first axis built
327 is the primary axis, and the other axis is the secondary axis.
328        ~~~~~~~~~~~~                            ~~~~~~~~~~~~~~
329 If the tab starts by building the vertical axis, then the
330 vertical axis is the primary axis, and the horizontal axis
331 is the secondary axis. And vice-versa.
332 
333 What's the difference between Horizontal & Vertical Tabs ?
334 
335   Structurally, V2x3 is the same as H3x2.  The only difference
336   is how Vim numbers the tabs, based on the axis that is the
337   primary axis.
338 
339   IF the primary axis is vertical, then numbering starts in
340   col 1 and goes down the windows in col 1, then to the top
341   of col 2, then down col 2, and so on.
342 
343   IF the primary axis is horizontal, numbering starts in
344   row 1 and goes left to right, then to row 2, left to right,
345   and so on.
346 
347   Most of the asymmetric tab builders defined in the plugin
348   use a vertical primary axis.  Developers can define any
349   mix of primary axis in their own tab builders -- See
350   "Adding New Builders"/TabwinsAddingNewBuilders
351

354 Section 4: 'Tabwins' Menu                               TabwinsMenu
355 ====================================================================
356 
357 In the 'Tabwins' menu built on plugin load, the extent that
358 populated tabs fill their buffers is dependent on the shell
359 vars available to the Vim session, and files/dirs present in
360 the local environment.
361 
362 The menu items reference some of these shell vars.
363 
364 The developer is encouraged to customize this menu for local usage.
365 
366 Update or replace
367   function Tabwins_menu_build()
368 in tabwins.vim to do so.

371 ====================================================================
372 Section 5: Configuration                      TabwinsConfiguration
373 ====================================================================
374 
375 The below globals are defined in tabwins.vim and are overridable.
376 
377 
378 To change Vertical/Horizontal max dimensions:
379   let g:tabwins_max_v_size                   = {N}
380   let g:tabwins_max_h_size                   = {N} 
381 Default: 15 for each
382 
383 
384 To limit tabwins.vim load to once per Vim session:
385   let g:tabwins_multiple_loads_are_permitted = 'N'
386 Default: 'Y'
387 
388 To prevent 'Tabwins' menu addition when tabwins loads:
389   let g:load_tabwins_menu_is_wanted          = 'N'
390 Default: 'Y'
391 
392 To change the 'Tabwins' menu number:
393   let g:tabwins_menu_number                  = {menu_number}
394 Default: 9998
395 
396 
397 To turn off creation of :Tabwins* aliases
398   let  g:tabwins_create_aliases_is_wanted    = 'N'
399 Default: 'Y'
400 => Creation of an alias command will NOT occur when an existing
401    command has the same name.
402 
403 :Tabwins* Alias Defaults:
404   let g:tabwins_alias            = 'T'
405   let g:tabwinsvertical_alias    = 'V'
406   let g:tabwinshorizontal_alias  = 'H'
407   let g:tabwinsfindtabwins_alias = 'TF'
408 
409 
410 To change the default beginning window number for filling fill_specs
411   let g:tabwins_first_window_number_default  = {N}
412 Default: 1
413 
414 To change the default window number that's active on :Tabwins* completion
415   let g:tabwins_ending_window_number_default = {N}
416 Default: 1
417 
418 
419 To change the default directory reader command:
420   let g:tabwins_dir_reader_name_default      = {dir_reader_command_name}
421 Default: 'Explore'
422 
423 To change the default top-line of netrw windows:
424   let g:tabwins_netrw_line_number_default    = {N}
425 Default: g:tabwins_netrw_line_number_dirpath, which defaults to 3
426 Calls to Fill_tab() can override this on a per-tab basis.
427 
428 To change the default netrw liststyle:
429   let g:tabwins_netrw_liststyle_default      = {N}
430 Default: 1
431 Calls to Fill_tab() can override this on a per-tab basis.
432 if netrw var g:netrw_liststyle is not defined, it gets this value.

436 Section 6: Adding New Builders            TabwinsAddingNewBuilders
437 ====================================================================
438 
439 
440 ADDING SYMMETRIC TAB COMMANDS --
441 
442   Bump either/both values of:
443     g:tabwins_max_v_size
444     g:tabwins_max_h_size
445   above the values assigned in tabwins.vim.
446 
447   Loading tabwins.vim will build commands for all the HxV and
448   VxH permutations.
449 
450   Of course, the set of symmetric tab builder commands can be
451   reduced by lowering the values of the same globals.
452 
453 
454 MORE GENERALLY --
455 
456   Adapt the below example to create your own commands that build tabs
457   that have the window structures and content that you want.
458 
459         let                g:home_parent_dirpath = substitute(finddir($HOME), '\/\w\+$', '', 'g')
460         command! TabwinsHomeDir :Tabwins 232;
461           \   $HOME,
462           \   'Explore ' . g:home_parent_dirpath . ' | resize 30',
463           \
464           \   'edit! ~/.bashrc',
465           \   '~/.gitignore',
466           \   'enew!',
467           \
468           \   '~/.vimrc',
469           \   '~/.vim'
470           \;
471           \ 'line_number_at_window_top' : 1
472 
473           " windows are filled with the order of elements
474           " in 'fill_specs' You want to spec something for each
475           " of the windows created by :Tabwins 232,
476           " either commands, filepaths, or dirpaths. Spec an empty
477           " string to bypass the fill of a window buffer.
478           " Specs can include shell or vim vars, and if files or dirs
479           " the spec can include a vim command or use the default :edit or
480           " :Explore call in Fill_tab()
481 
482           " NOTE-the .vim file implementation defines 'home_parent_dirpath' as
483           " as a vimscript var, i.e. 's:home_parent_dirpath' -- it's changed
484           " here to facilitate interactive experimentation.
485 
486 
487   See the other demo commands at end of the tabwins.vim.
488 
489   See HELP in TabwinsUsage.
